题目

爬梯子

假设你正在爬楼梯。需要 n 阶你才能到达楼顶。每次你可以爬 1 或 2 个台阶 你有多少种不同的方法可以爬到楼顶呢？（给定 n 是一个正整数）

  • 递归
function step(i) {
  if (i === 1) return 1;
  if (i === 2) return 1;
  return step(i - 1) + step(i - 2);
}

step(10);
  • 动态规划
const cache: number[] = [];

function step(i: number) {
  if (i === 1) return (cache[1] = 1);
  if (i === 2) return (cache[2] = 1);
  // 子路径缓存
  return (cache[i] = cache[i - 1] + cache[i - 2]);
}

// 从低开始，利用缓存
for (let i = 1; i <= 10; i++) step(i);

cache[10];
